class ParkingSystem {
    // Array to hold the remaining slots for each car type.
    // Index 0 is unused, index 1 = big, index 2 = medium, index 3 = small.
    private int[] slots;

    public ParkingSystem(int big, int medium, int small) {
        // We initialize the array with size 4 so we can use carType (1, 2, 3) 
        // directly as the array index without doing any index shifts (like carType - 1).
        this.slots = new int[]{0, big, medium, small};
    }
    
    public boolean addCar(int carType) {
        // Check if there is a slot available for this specific carType
        if (this.slots[carType] > 0) {
            // Decrement the available slots since a car just parked
            this.slots[carType]--;
            return true;
        }
        // No slots left for this car type
        return false;
    }
}
